\newcommand{\categoryformat}[1]{{\color{darkgray} \footnotesize #1}}

\newcommand{\institutionformat}[1]{{\fontfamily{cmss}\selectfont \textbf{#1}}}

\newcommand{\dateformat}[1]{{\small (#1)}}

\newcommand{\descriptionformat}[1]{
\begin{adjustwidth}{5mm}{5mm}
{\small #1}
\end{adjustwidth}
}

\newcommand{\commonformat}[4]{
\vspace{2mm}
\categoryformat{#1}

\institutionformat{#2} \dateformat{#3}

\descriptionformat{#4}
}

\newcommand{\skillformat}[3]{
\vspace{2mm}
\categoryformat{#1}

\institutionformat{#2}

\descriptionformat{#3}
}

\newcommand{\readwithcommonformat}[1]{
\csvreader[common]{#1}{}{
\commonformat{\shortdesc}{\institution}{\dates}{\description}
}
}

\newcommand{\readwithskillformat}[1]{
\csvreader[common]{#1}{}{
\skillformat{\category}{\header}{\description}
}
}
